**1. Three tone template functions** — each is a pure function `(name, company, pain, value)` returning a string. These must produce genuinely distinct tones and stay under 100 words by construction (aim 65–80 words, hard ceiling 99):

```javascript
function directTemplate(name, company, pain, value) {
  // Direct tone: declarative statement of fact, no question, hard CTA
  // Structure: greeting → fact about their pain → how long it costs them → your solution + timeframe → CTA
  return `Hi ${name},

I noticed ${company} is likely dealing with ${pain}. Most teams I speak with lose hours every week to this.

${value} — and we typically help teams fix it in under 30 days.

Worth a quick 15-minute call this week?`;
}

function curiousTemplate(name, company, pain, value) {
  // Curious tone: opens WITH a question, withholds resolution, soft close
  // Structure: greeting → direct question about pain → we've helped with value → soft offer
  return `Hi ${name},

Quick question — is ${company} actively working on ${pain} right now, or is it on the backburner?

I ask because we've helped similar teams tackle it using ${value}.

Happy to share what worked — would that be useful?`;
}

function storyTemplate(name, company, pain, value) {
  // Story-Led tone: micro-narrative opener, connects to their situation
  // Structure: greeting → "A [role] at a company like yours..." → what they did (value) → result → CTA
  return `Hi ${name},

A sales leader at a company like ${company} came to us struggling with exactly ${pain}.

They tried ${value} and turned it around in one quarter.

I'd love to share the specifics — worth 15 minutes?`;
}
```

**3. generate() function:**
```javascript
function generate() {
  const name    = document.getElementById('inp-name').value.trim()    || 'there';
  const company = document.getElementById('inp-company').value.trim() || 'your company';
  const pain    = document.getElementById('inp-pain').value.trim()    || 'your key challenge';
  const value   = document.getElementById('inp-value').value.trim()   || 'our solution';

  const direct  = directTemplate(name, company, pain, value);
  const curious = curiousTemplate(name, company, pain, value);
  const story   = storyTemplate(name, company, pain, value);

  renderVariant('card-direct',  direct);
  renderVariant('card-curious', curious);
  renderVariant('card-story',   story);

  const results = document.getElementById('results');
  results.style.display = 'block';
  results.classList.remove('fade-in');
  void results.offsetWidth; // force reflow to retrigger animation
  results.classList.add('fade-in');
  results.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
}
```

**4. renderVariant() function** — uses textContent, never innerHTML for user-provided values (XSS safety):
```javascript
function renderVariant(cardId, body) {
  const card = document.getElementById(cardId);
  card.querySelector('.variant-body').textContent = body;
  card.querySelector('.copy-btn').dataset.body = body;
}
```

**5. copyBody() function** — includes file:// fallback per research pitfall #3:
```javascript
function copyBody(btn) {
  const text = btn.dataset.body;
  if (!text) return;
  const orig = btn.textContent;
  navigator.clipboard.writeText(text).then(() => {
    btn.textContent = 'Copied!';
    btn.classList.add('copied');
    setTimeout(() => { btn.textContent = orig; btn.classList.remove('copied'); }, 1500);
  }).catch(() => {
    // Fallback for file:// protocol in Firefox
    const ta = document.createElement('textarea');
    ta.value = text;
    document.body.appendChild(ta);
    ta.select();
    document.execCommand('copy');
    document.body.removeChild(ta);
    btn.textContent = 'Copied!';
    btn.classList.add('copied');
    setTimeout(() => { btn.textContent = orig; btn.classList.remove('copied'); }, 1500);
  });
}
```

function resetForm() {

}
